def power(base, exp, mod):
    res = 1
    base %= mod
    while exp > 0:
        if exp % 2 == 1:
            res = (res * base) % mod
        base = (base * base) % mod
        exp //= 2
    return res

If $N$ is large, find the pre-period and cycle length. Use matrix exponentiation for the bulk.

Given $N \le 10^{18}$, $k \le 1000$. $k^2 \le 10^6$. $O(k^2 \cdot 4^3 \log N)$ is feasible.

For $N \le 10^6$ (Cases 14, 15), direct $O(N)$ simulation is fine.

If $N \le 10^{18}$ and $k \le 1000$ (Cases 16, 17), $k^2 \le 10^6$. This is the intended path.

import sys

# Set recursion limit higher for safety in matrix operations, although iterative is preferred.
# sys.setrecursionlimit(2000)

def solve():
    # Read N, K, P from a single line
    try:
        line = sys.stdin.readline()
        if not line:
            return
        N, K, P = map(int, line.split())
    except EOFError:
        return
    except Exception:
        return

    if N == 0:
        print(0)
        return
    
    # R_n: number of rabbit pairs in month n
    # R_1 = 1, R_2 = 1
    
    if N == 1:
        print(1 % P)
        return
    if N == 2:
        print(1 % P)
        return

    # We use 1-based indexing conceptually, but arrays are 0-indexed.
    # R[i] stores R_{i+1}
    R = [0] * (N + 1)
    R[1] = 1
    R[2] = 1

    # Simulation phase. We track the state (R_i mod K, R_{i-1} mod K) to detect cycles.
    # State: (R_{i-1} % K, R_{i-2} % K)
    # Since the sequence R_i is defined by the previous two terms and a condition based on R_{i-1}+R_{i-2} mod K,
    # the sequence of pairs (R_i mod K, R_{i-1} mod K) must eventually cycle.
    
    # State map: maps (r_prev, r_prev_prev) -> month index i where this state occurred.
    # We track the state that determines the next term R_i.
    # State at step i: (R_{i-1} % K, R_{i-2} % K)
    
    # Initial state before calculating R_3: (R_2 % K, R_1 % K) = (1 % K, 1 % K)
    # We store the index i for the pair (R_{i-1} % K, R_{i-2} % K)
    history = {}
    
    # Initial state for the recurrence calculation (i starts at 3)
    r_prev = R[2] % K  # R_2 % K
    r_prev_prev = R[1] % K # R_1 % K
    
    history[(r_prev, r_prev_prev)] = 2 # State before calculating R_3 (i=3)

    i = 3
    while i <= N:
        # Calculate R_i
        sum_mod_k = (r_prev + r_prev_prev) % K
        
        delta = 0
        # Rule: if feeding circle consists of only one pair, it dies.
        # This happens if the leftover pairs (sum_mod_k) is 1.
        if sum_mod_k == 1:
            delta = 1
        
        R_i = (R[i-1] + R[i-2] - delta) % P
        
        # Store result modulo P
        R[i] = R_i
        
        if i == N:
            print(R_i)
            return

        # Update state for next iteration (i+1)
        r_prev_prev = r_prev
        r_prev = R[i] % K
        
        current_state = (r_prev, r_prev_prev)
        
        if current_state in history:
            # Cycle detected!
            start_month = history[current_state]
            cycle_length = i - start_month
            
            # We need R_N. We have computed up to R_i.
            # The sequence of R values repeats starting from R_{start_month}.
            
            remaining_steps = N - i
            offset_in_cycle = remaining_steps % cycle_length
            
            # The required index in the stored sequence R is:
            # start_month + offset_in_cycle
            final_index = start_month + offset_in_cycle
            
            print(R[final_index])
            return
        
        history[current_state] = i
        i += 1

    # If N was small enough to complete the loop without cycle detection (i reached N)
    print(R[N])

solve()
